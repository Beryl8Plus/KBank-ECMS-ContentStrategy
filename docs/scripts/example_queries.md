# ดึงข้อมูล Rule พร้อมเงื่อนไขและค่า Setting ทั้งหมด เพื่อนำไปคำนวณที่ระบบหลังบ้าน
```sql
SELECT 
    dr.NAME AS DECISION_RULE_NAME,
    dr.TYPE AS RULE_TYPE,
    r.VARIATION_NAME,
    rc.SEQUENCE,
    attr.FIELD_NAME AS ATTRIBUTE_NAME,
    rc.LOGICAL_OPERATOR,
    ra.VALUE AS VARIATION_VALUE,
    rc.CONNECTOR_OPERATOR
FROM DECISION_RULES dr
JOIN RULES r ON dr.DECISION_RULE_ID = r.DECISION_RULE_ID
JOIN RULE_CONDITIONS rc ON dr.DECISION_RULE_ID = rc.DECISION_RULE_ID
JOIN ATTRIBUTES attr ON rc.ATTRIBUTE_ID = attr.ATTRIBUTE_ID
LEFT JOIN RULE_ATTRIBUTES ra ON r.RULE_ID = ra.RULE_ID AND attr.ATTRIBUTE_ID = ra.ATTRIBUTE_ID
WHERE dr.STATUS = 'ACTIVE' 
ORDER BY dr.NAME, r.ORDER_NO, rc.SEQUENCE;
```
