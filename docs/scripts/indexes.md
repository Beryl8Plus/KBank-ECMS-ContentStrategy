# Indexes (สำหรับเพิ่ม Performance และ Unique Constraints)
```sql
CREATE UNIQUE INDEX IDX_PERM_FEATURE_ACTION ON PERMISSIONS (FEATURE_CODE, ACTION);
CREATE UNIQUE INDEX IDX_PROFILE_PERM_UNIQUE ON PROFILE_PERMISSIONS (PROFILE_ID, PERMISSION_ID);
CREATE UNIQUE INDEX IDX_SCH_DR_PL_UNIQUE ON SCHEDULES (DECISION_RULE_ID, PLACEMENT_ID);

CREATE INDEX IDX_USERS_EMAIL ON USERS (EMAIL);
CREATE INDEX IDX_DR_STATUS ON DECISION_RULES (STATUS);
CREATE INDEX IDX_RULES_DR_ID ON RULES (DECISION_RULE_ID);
CREATE INDEX IDX_SCH_EFFECTIVE ON SCHEDULES (EFFECTIVE_FROM, EFFECTIVE_UNTIL, IS_ACTIVE);
CREATE INDEX IDX_OCCURRENCE_START ON SCHEDULE_OCCURRENCES (OCCURRENCE_START);
```
